## 二、日志规范

### 2.1 日志级别定义

| 级别 | 含义 | 使用场景 | 输出 |
|------|------|----------|------|
| **VERBOSE** | 详细 | 开发调试 | Debug build |
| **DEBUG** | 调试 | 开发调试 | Debug build |
| **INFO** | 信息 | 关键业务流程 | 所有环境 |
| **WARNING** | 警告 | 可恢复的异常 | 所有环境 |
| **ERROR** | 错误 | 需要关注的异常 | 所有环境 |
| **ASSERT** | 严重 | 不应该发生 | 所有环境 |

### 2.2 日志格式规范

```kotlin
// ✅ 正确：结构化日志
Timber.tag("MorningCheck")
    .d("Face recognized: userId=$userId, name=$name, confidence=$confidence")

Timber.tag("Record")

Timber.tag("Error")
    .e(exception, "Face recognition failed: $errorMessage")
```


### 4.2 性能统计类

```kotlin
/**
 * 性能指标记录器
 */
object PerformanceMetrics {
    
    private val metrics = mutableMapOf<String, MetricRecord>()
    
    /**
     * 记录操作耗时
     */
    fun recordDuration(operation: String, durationMs: Long) {
        val record = metrics.getOrPut(operation) { MetricRecord(operation) }
        record.addSample(durationMs)
        
        Timber.tag("Perf")
            .d("$operation: ${durationMs}ms (avg: ${record.averageMs}ms)")
    }
    
    /**
     * 获取统计信息
     */
    fun getStats(operation: String): MetricStats? {
        return metrics[operation]?.getStats()
    }
}

data class MetricRecord(
    val operation: String,
    private val samples = mutableListOf<Long>()
) {
    fun addSample(durationMs: Long) {
        synchronized(samples) {
            samples.add(durationMs)
            // 只保留最近1000条
            if (samples.size > 1000) {
                samples.removeAt(0)
            }
        }
    }
    
    fun getStats(): MetricStats {
        synchronized(samples) {
            if (samples.isEmpty()) return MetricStats(0, 0, 0, 0)
            
            val sorted = samples.sorted()
            return MetricStats(
                count = samples.size,
                avg = samples.average().toLong(),
                min = sorted.first(),
                max = sorted.last(),
                p50 = sorted[sorted.size / 2],
                p95 = sorted[(sorted.size * 0.95).toInt()],
                p99 = sorted[(sorted.size * 0.99).toInt()]
            )
        }
    }
}

data class MetricStats(
    val count: Int,
    val avg: Long,
    val min: Long,
    val max: Long,
    val p50: Long = 0,
    val p95: Long = 0,
    val p99: Long = 0
)
```

### 4.3 使用示例

```kotlin
class FaceRecognitionUseCase(
    private val faceService: IFaceRecognitionService
) {
    
    suspend fun recognize(bitmap: Bitmap): Result<User> {
        val startTime = System.currentTimeMillis()
        
        try {
            val result = faceService.detectAndCompare(bitmap)
            
            // 记录性能
            val duration = System.currentTimeMillis() - startTime
            PerformanceMetrics.recordDuration("face_recognition", duration)
            
            return result
        } catch (e: Exception) {
            Timber.tag("Perf")
                .e("Face recognition failed after ${System.currentTimeMillis() - startTime}ms")
            throw e
        }
    }
}
```

### 5.2 追踪实现

```kotlin
/**
 * 用户操作追踪器
 */
object UserActionTracker {
    
    private val logs = mutableListOf<UserActionLog>()
    
    fun track(
        action: ActionType,
        screen: String,
        detail: String = "",
        result: ActionResult = ActionResult.SUCCESS,
        durationMs: Long? = null
    ) {
        val log = UserActionLog(
            timestamp = System.currentTimeMillis(),
            userId = getCurrentUserId(),
            adminId = getCurrentAdminId(),
            action = action,
            screen = screen,
            detail = detail,
            result = result,
            durationMs = durationMs
        )
        
        synchronized(logs) {
            logs.add(log)
        }
        
        // 输出日志
        Timber.tag("Action")
            .d("$action on $screen: $detail [${result.name}]")
    }
    
    fun getRecentLogs(count: Int = 100): List<UserActionLog> {
        synchronized(logs) {
            return logs.takeLast(count)
        }
    }
}
```

### 6.2 状态监控类

```kotlin
/**
 * 系统状态监控器
 */
object SystemMonitor {
    
    private val _systemState = MutableStateFlow(SystemState())
    val systemState: StateFlow<SystemState> = _systemState.asStateFlow()
    
    data class SystemState(
        val memoryUsage: Float = 0f,           // 内存使用率 0.0-1.0
        val availableStorage: Long = 0L,       // 可用存储 bytes
        val databaseSize: Long = 0L,          // 数据库大小 bytes
        val recordCount: Int = 0,              // 记录数量
        val aiModelLoaded: Boolean = false,   // AI模型状态
        val cameraReady: Boolean = false,      // 摄像头状态
        val temperatureReady: Boolean = false, // 测温模块状态
        val lastHealthCheck: Long = 0L         // 最后健康检查时间
    )
    
    /**
     * 执行健康检查
     */
    fun performHealthCheck(): List<HealthIssue> {
        val issues = mutableListOf<HealthIssue>()
        
        // 检查内存
        val runtime = Runtime.getRuntime()
        val usedMemory = runtime.totalMemory() - runtime.freeMemory()
        val maxMemory = runtime.maxMemory()
        val memoryUsage = usedMemory.toFloat() / maxMemory
        
        _systemState.update { it.copy(memoryUsage = memoryUsage) }
        
        if (memoryUsage > 0.8f) {
            issues.add(HealthIssue(
                type = IssueType.MEMORY_HIGH,
                severity = Severity.WARNING,
                message = "内存使用率 ${(memoryUsage * 100).toInt()}%"
            ))
        }
        
        // 检查存储
        val availableStorage = getAvailableStorage()
        _systemState.update { it.copy(availableStorage = availableStorage) }
        
        if (availableStorage < 100 * 1024 * 1024) {
            issues.add(HealthIssue(
                type = IssueType.STORAGE_LOW,
                severity = Severity.ERROR,
                message = "存储空间不足 ${formatSize(availableStorage)}"
            ))
        }
        
        // 检查数据库
        val dbSize = getDatabaseSize()
        _systemState.update { it.copy(databaseSize = dbSize) }
        
        if (dbSize > 500 * 1024 * 1024) {
            issues.add(HealthIssue(
                type = IssueType.DATABASE_LARGE,
                severity = Severity.WARNING,
                message = "数据库过大 ${formatSize(dbSize)}"
            ))
        }
        
        _systemState.update { it.copy(lastHealthCheck = System.currentTimeMillis()) }
        
        return issues
    }
}

data class HealthIssue(
    val type: IssueType,
    val severity: Severity,
    val message: String
)

enum class IssueType {
    MEMORY_HIGH,
    STORAGE_LOW,
    DATABASE_LARGE,
    AI_MODEL_FAILED,
    CAMERA_UNAVAILABLE,
    TEMPERATURE_UNAVAILABLE
}

enum class Severity {
    INFO,
    WARNING,
    ERROR
}
```

```kotlin
/**
 * 日志文件管理
 */
object LogFileManager {
    
    private const val MAX_LOG_SIZE = 10 * 1024 * 1024  // 10MB
    private const val MAX_LOG_FILES = 5
    
    /**
     * 初始化日志目录
     */
    fun init(context: Context) {
        val logDir = File(context.filesDir, "logs")
        if (!logDir.exists()) {
            logDir.mkdirs()
        }
    }
    
    /**
     * 清理过期日志
     */
    fun cleanOldLogs() {
        val logDir = getLogDir()
        val files = logDir.listFiles()?.filter { it.extension == "log" } ?: return
        
        // 按修改时间排序
        val sortedFiles = files.sortedByDescending { it.lastModified() }
        
        // 删除旧文件
        sortedFiles.drop(MAX_LOG_FILES).forEach { it.delete() }
        
        // 删除过期文件（7天前）
        val sevenDaysAgo = System.currentTimeMillis() - 7 * 24 * 60 * 60 * 1000L
        sortedFiles.forEach { file ->
            if (file.lastModified() < sevenDaysAgo) {
                file.delete()
            }
        }
    }
    
    /**
     * 获取日志文件列表
     */
    fun getLogFiles(): List<LogFileInfo> {
        return getLogDir().listFiles()
            ?.filter { it.extension == "log" }
            ?.sortedByDescending { it.lastModified() }
            ?.map { LogFileInfo(it.name, it.length(), it.lastModified()) } ?: emptyList()
    }
}

data class LogFileInfo(
    val name: String,
    val size: Long,
    val modifiedAt: Long
)
```

### 8.1 本地日志分析

```kotlin
/**
 * 日志分析器
 */
object LogAnalyzer {
    
    /**
     * 分析错误日志
     */
    fun analyzeErrors(logs: List<ErrorLog>): ErrorAnalysis {
        val errorCount = logs.size
        val errorTypes = logs.groupBy { it.tag }
            .mapValues { it.value.size }
            .toList()
            .sortedByDescending { it.second }
        
        return ErrorAnalysis(
            totalErrors = errorCount,
            errorByType = errorTypes.take(10),
            recentErrors = logs.takeLast(10)
        )
    }
    
    /**
     * 分析性能数据
     */
    fun analyzePerformance(operation: String): PerformanceAnalysis {
        val stats = PerformanceMetrics.getStats(operation) ?: return PerformanceAnalysis()
        
        return PerformanceAnalysis(
            operation = operation,
            avgDuration = stats.avg,
            minDuration = stats.min,
            maxDuration = stats.max,
            p50 = stats.p50,
            p95 = stats.p95,
            p99 = stats.p99,
            count = stats.count
        )
    }
    
    /**
     * 生成报告
     */
    fun generateReport(): String {
        return buildString {
            appendLine("=== 系统日志报告 ===")
            appendLine("生成时间: ${Date()}")
            appendLine()
            
            appendLine("--- 错误统计 ---")
            PerformanceMetrics.getStats("face_recognition")?.let {
                appendLine("人脸识别: 平均 ${it.avg}ms, P95 ${it.p95}ms")
            }
            PerformanceMetrics.getStats("hand_check")?.let {
                appendLine("手部检测: 平均 ${it.avg}ms, P95 ${it.p95}ms")
            }
            appendLine()
            
            appendLine("--- 系统状态 ---")
            val state = SystemMonitor.systemState.value
            appendLine("内存使用: ${(state.memoryUsage * 100).toInt()}%")
            appendLine("数据库大小: ${formatSize(state.databaseSize)}")
            appendLine("记录数量: ${state.recordCount}")
        }
    }
}
```

# 项目重构实施计划

---

## 文档说明

本文档是 SmartCheck AI 项目的重构实施计划，详细描述了从当前架构迁移到新架构的每个步骤、任务和验收标准。

---

## 一、重构目标

### 1.1 当前问题

| 问题 | 现状 | 影响 |
|------|------|------|
| ViewModel 职责过重 | MainViewModel 953 行 | 难以维护、无法测试 |
| Repository 无接口 | 直接依赖具体类 | 无法 Mock、单元测试困难 |
| 业务逻辑散落 | 代码重复 | 难以复用 |
| 状态定义混乱 | 散落代码各处 | 维护困难 |

### 1.2 目标架构

```
┌─────────────────────────────────────────────────────────┐
│                    新架构（目标）                        │
├─────────────────────────────────────────────────────────┤
│  UI Layer    →  ViewModel (变薄，专注 UI 状态)        │
├─────────────────────────────────────────────────────────┤
│  UseCase Layer → 业务逻辑封装，可复用、可测试          │
├─────────────────────────────────────────────────────────┤
│  Repository Layer → 接口 + 实现分离                    │
├─────────────────────────────────────────────────────────┤
│  Data Layer    → Room、File、Serial、JNI              │
└─────────────────────────────────────────────────────────┘
```

---

## 二、重构范围

### 2.1 涉及模块

| 模块 | 重构内容 | 优先级 |
|------|----------|--------|
| domain/ | 新建 Domain Layer | P0 |
| data/repository/ | 添加接口 | P0 |
| viewmodel/ | 依赖注入改为接口 | P0 |
| usecase/ | 新建业务逻辑层 | P1 |

### 2.2 不涉及

| 模块 | 原因 |
|------|------|
| UI (screens/components) | 界面已有，保持不变 |
| 数据库 (db/) | Room Entity/DAO 已有 |
| 硬件相关 (serial/) | 已有实现 |

**文件**：`domain/model/User.kt`

```kotlin
package com.smartcheck.app.domain.model

/**
 * 员工领域模型
 */
data class User(
    val id: Long = 0,
    val name: String,
    val employeeId: String,
    val idCardNumber: String = "",
    val faceImagePath: String? = null,
    val faceEmbedding: ByteArray? = null,
    val healthCertImagePath: String = "",
    val healthCertStartDate: Long? = null,
    val healthCertEndDate: Long? = null,
    val department: String = "",
    val isActive: Boolean = true,
    val createdAt: Long = System.currentTimeMillis()
) {
    /**
     * 计算健康证状态
     */
    fun getHealthCertStatus(): HealthCertStatus {
        val endDate = healthCertEndDate ?: return HealthCertStatus.EXPIRED
        val now = System.currentTimeMillis()
        val daysRemaining = (endDate - now) / (24 * 60 * 60 * 1000)
        
        return when {
            daysRemaining < 0 -> HealthCertStatus.EXPIRED
            daysRemaining <= 7 -> HealthCertStatus.EXPIRING_SOON
            else -> HealthCertStatus.VALID
        }
    }
    
    /**
     * 健康证剩余天数
     */
    fun getHealthCertDaysRemaining(): Long? {
        val endDate = healthCertEndDate ?: return null
        val now = System.currentTimeMillis()
        return (endDate - now) / (24 * 60 * 60 * 1000)
    }
}
```

```kotlin
package com.smartcheck.app.domain.usecase

import com.smartcheck.app.domain.model.AppError
import com.smartcheck.app.domain.model.User
import com.smartcheck.app.domain.repository.IUserRepository
import kotlinx.coroutines.flow.Flow

/**
 * 员工管理 UseCase
 */
class UserManageUseCase(
    private val userRepository: IUserRepository
) {
    
    /**
     * 观察所有员工
     */
    fun observeAllUsers(): Flow<List<User>> {
        return userRepository.observeAllUsers()
    }
    
    /**
     * 获取员工详情
     */
    suspend fun getUserById(id: Long): Result<User, AppError> {
        return userRepository.getUserById(id)
    }
    
    /**
     * 创建员工
     */
    suspend fun createUser(user: User): Result<Long, AppError> {
        // 验证
        if (user.name.isBlank()) {
            return Result.failure(AppError.ValidationError("name", "姓名不能为空"))
        }
        if (user.employeeId.isBlank()) {
            return Result.failure(AppError.ValidationError("employeeId", "工号不能为空"))
        }
        
        // 检查工号是否存在
        val existing = userRepository.getUserByEmployeeId(user.employeeId)
        if (existing.isSuccess) {
            return Result.failure(AppError.DuplicateError("employeeId", user.employeeId))
        }
        
        return userRepository.createUser(user)
    }
    
    /**
     * 更新员工
     */
    suspend fun updateUser(user: User): Result<Unit, AppError> {
        // 验证
        if (user.name.isBlank()) {
            return Result.failure(AppError.ValidationError("name", "姓名不能为空"))
        }
        
        return userRepository.updateUser(user)
    }
    
    /**
     * 删除员工（软删除）
     */
    suspend fun deleteUser(userId: Long): Result<Unit, AppError> {
        return userRepository.deleteUser(userId)
    }
}
```

```kotlin
class MainViewModel @Inject constructor(
    private val morningCheckUseCase: MorningCheckUseCase,
    private val userManageUseCase: UserManageUseCase,
    private val voiceService: IVoiceService
) : ViewModel() {
    
    private val _uiState = MutableStateFlow(HomeUiState())
    val uiState: StateFlow<HomeUiState> = _uiState.asStateFlow()
    
    fun onEvent(event: HomeEvent) {
        when (event) {
            is HomeEvent.StartCheck -> startCheck(event.bitmap)
            is HomeEvent.MeasureTemp -> measureTemperature()
            is HomeEvent.CheckHand -> checkHand(event.bitmap)
            // ...
        }
    }
    
    private fun startCheck(bitmap: Bitmap) {
        viewModelScope.launch {
            val result = morningCheckUseCase.startCheck(bitmap)
            result.onSuccess { checkInResult ->
                _uiState.update { it.copy(
                    user = checkInResult.user,
                    healthCertStatus = checkInResult.healthCertStatus
                )}
            }.onFailure { error ->
                _uiState.update { it.copy(error = error) }
            }
        }
    }
}
```

```kotlin
import io.mockk.coVerify
import io.mockk.mockk
import kotlinx.coroutines.flow.flowOf
import kotlinx.coroutines.test.runTest
import org.junit.Assert.*
import org.junit.Before
import org.junit.Test

class UserManageUseCaseTest {
    
    private lateinit var userRepository: IUserRepository
    private lateinit var useCase: UserManageUseCase
    
    @Before
    fun setup() {
        userRepository = mockk()
        useCase = UserManageUseCase(userRepository)
    }
    
    @Test
    fun `createUser with valid data should succeed`() = runTest {
        // Given
        val user = User(name = "张三", employeeId = "001")
        coEvery { userRepository.getUserByEmployeeId("001") } returns Result.failure(AppError.NotFound)
        coEvery { userRepository.createUser(user) } returns Result.success(1L)
        
        // When
        val result = useCase.createUser(user)
        
        // Then
        assertTrue(result.isSuccess)
        assertEquals(1L, result.getOrNull())
        coVerify { userRepository.createUser(user) }
    }
    
    @Test
    fun `createUser with blank name should fail`() = runTest {
        // Given
        val user = User(name = "", employeeId = "001")
        
        // When
        val result = useCase.createUser(user)
        
        // Then
        assertTrue(result.isFailure)
        assertTrue(result.exceptionOrNull() is AppError.ValidationError)
    }
    
    @Test
    fun `createUser with duplicate employeeId should fail`() = runTest {
        // Given
        val existingUser = User(id = 1, name = "李四", employeeId = "001")
        val newUser = User(name = "张三", employeeId = "001")
        coEvery { userRepository.getUserByEmployeeId("001") } returns Result.success(existingUser)
        
        // When
        val result = useCase.createUser(newUser)
        
        // Then
        assertTrue(result.isFailure)
        assertTrue(result.exceptionOrNull() is AppError.DuplicateError)
    }
}
```
